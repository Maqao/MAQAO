ADD_CUSTOM_TARGET(doc)



# ---- Generate the C documentation using Doxygen
if (${IS_DOXYGEN})
   CONFIGURE_FILE(${CMAKE_CURRENT_SOURCE_DIR}/internals/Doxyfile.in 
                  ${CMAKE_CURRENT_SOURCE_DIR}/internals/Doxyfile @ONLY)
   CONFIGURE_FILE(${CMAKE_CURRENT_SOURCE_DIR}/internals/CoreDeveloperGuide.html.in 
                  ${CMAKE_CURRENT_SOURCE_DIR}/internals/CoreDeveloperGuide.html @ONLY)
                  
   ADD_CUSTOM_TARGET (doc_doxygen
            COMMAND ${CMAKE_COMMAND} -E copy ${CMAKE_CURRENT_SOURCE_DIR}/internals/CoreDeveloperGuide.html ${CMAKE_CURRENT_SOURCE_DIR}/
            COMMAND ${DOXYGEN_BIN} ${CMAKE_CURRENT_SOURCE_DIR}/internals/Doxyfile
            COMMAND ${CMAKE_COMMAND} -E copy_directory ${CMAKE_CURRENT_SOURCE_DIR}/html ${CMAKE_CURRENT_SOURCE_DIR}/internals/html
            COMMAND ${CMAKE_COMMAND} -E remove_directory ${CMAKE_CURRENT_SOURCE_DIR}/html
            COMMAND ${CMAKE_COMMAND} -E copy_directory ${CMAKE_CURRENT_SOURCE_DIR}/latex ${CMAKE_CURRENT_SOURCE_DIR}/internals/latex
            COMMAND ${CMAKE_COMMAND} -E remove_directory ${CMAKE_CURRENT_SOURCE_DIR}/latex
            )
   ADD_DEPENDENCIES (doc doc_doxygen)
endif (${IS_DOXYGEN})

# ---- Generate the LUA documentation using luadoc
if (${IS_LUADOC})
   CONFIGURE_FILE(${CMAKE_CURRENT_SOURCE_DIR}/internals/DeveloperGuide.html.in 
                  ${CMAKE_CURRENT_SOURCE_DIR}/internals/DeveloperGuide.html @ONLY)
   ADD_CUSTOM_TARGET (doc_luadoc
            COMMAND ${CMAKE_COMMAND} -E copy ${CMAKE_CURRENT_SOURCE_DIR}/internals/DeveloperGuide.html ${CMAKE_CURRENT_SOURCE_DIR}/
            COMMAND ${LUADOC_BIN} -d ${CMAKE_CURRENT_SOURCE_DIR}/internals/lua_API 
                                     ${CMAKE_SOURCE_DIR}/src/plugins/built_in/abstract_objects/stub/ao_*.luadoc
                                     ${CMAKE_SOURCE_DIR}/src/plugins/built_in/classes/stub/cl_*.luadoc
            )
   ADD_DEPENDENCIES (doc doc_luadoc)
endif (${IS_LUADOC})

### --- For cleaning files generated by cmake --- ###
ADD_CUSTOM_TARGET(distclean_doc
    COMMAND ${CMAKE_COMMAND} -E remove ${CMAKE_CURRENT_SOURCE_DIR}/CoreDeveloperGuide.html
    COMMAND ${CMAKE_COMMAND} -E remove ${CMAKE_CURRENT_SOURCE_DIR}/DeveloperGuide.html
    COMMAND ${CMAKE_COMMAND} -E remove ${CMAKE_CURRENT_SOURCE_DIR}/internals/Doxyfile
    COMMAND ${CMAKE_COMMAND} -E remove ${CMAKE_CURRENT_SOURCE_DIR}/internals/CoreDeveloperGuide.html
    COMMAND ${CMAKE_COMMAND} -E remove ${CMAKE_CURRENT_SOURCE_DIR}/internals/DeveloperGuide.html
    COMMAND ${CMAKE_COMMAND} -E remove ${CMAKE_CURRENT_SOURCE_DIR}/internals/doxygen.errors
    COMMAND ${CMAKE_COMMAND} -E remove_directory ${CMAKE_CURRENT_SOURCE_DIR}/internals/html
    COMMAND ${CMAKE_COMMAND} -E remove_directory ${CMAKE_CURRENT_SOURCE_DIR}/internals/latex
    COMMAND ${CMAKE_COMMAND} -E remove_directory ${CMAKE_CURRENT_SOURCE_DIR}/internals/lua_API
    VERBATIM
    )
